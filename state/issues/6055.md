## Description

# 创建插件目录（若不存在）
mkdir -p ~/.openclaw/plugins

# 克隆插件源码到指定目录
git clone https://github.com/m1heng/Clawdbot-feishu.git ~/.openclaw/plugins/@m1heng-clawd/feishu

# 强制清空环境变量
unset NODE_OPTIONS
export NODE_OPTIONS=""

# 安装插件（全程无环境变量干扰）
cd ~/.openclaw/plugins/@m1heng-clawd/feishu
openclaw plugins install .
